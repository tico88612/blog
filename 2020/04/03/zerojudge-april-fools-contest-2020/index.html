string decToBin(int s){
    string re;
    REP(i, 4){
        if(s % 2){
            re += '1';
        }
        else{
            re += '0';
        }
        s /= 2;
    }
    reverse(ALL(re));
    return re;
}

string hexToBin(char s){
    int ans = 0;
    if (isdigit(s)){
        ans = s - '0';
    }
    else {
        ans = s - 'a' + 10;
    }
    return decToBin(ans);
}

char decToHex(int s) {
    if (s < 10) {
        return s + '0';
    }
    else {
        return s - 10 + 'a';
    }
}

char binToHex(string s){
    int total = 0;
    int now = 8, i = 0;
    while (now) {
        if (s[i] == '1') {
            total += now;
        }
        i++;
        now /= 2;
    }
    return decToHex(total);
}

ll q_pow(ll a, ll n, ll MOD) {
    ll res = 1;
    while (n) {
        if (n & 1) res = res * a % MOD;
        a = a * a % MOD;
        n >>= 1;
    }
    return res % MOD;
}

ll sum(ll n) {
    if (n <= 1) return a % MOD;

    ll s = sum(n / 2) % MOD;

    if (n % 2 == 0)
        return s + s * q_pow(a, n / 2, MOD) % MOD;
    else
        return s + s * q_pow(a, n / 2, MOD) + q_pow(a, n, MOD) % MOD;
}
